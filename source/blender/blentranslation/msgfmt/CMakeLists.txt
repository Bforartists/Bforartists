# Build msgfmt executable

set(INC
  ../../../../intern/guardedalloc
  ../../blenlib
)

set(SRC
  msgfmt.c
)

set(LIB
  bf_blenlib
  bf_intern_guardedalloc
  ${ZLIB_LIBRARIES}

target_link_libraries(msgfmt ${LIB})
